-- 2. Create the unified notification settings schema matrix
CREATE TABLE public.notification_settings (
    id INT PRIMARY KEY DEFAULT 1,
    min_temp NUMERIC NOT NULL DEFAULT 15.0,
    max_temp NUMERIC NOT NULL DEFAULT 28.5,
    min_hum NUMERIC NOT NULL DEFAULT 70.0,
    max_hum NUMERIC NOT NULL DEFAULT 95.0,
    min_co2 INT NOT NULL DEFAULT 400,
    max_co2 INT NOT NULL DEFAULT 1800,
    email_address VARCHAR(255) NOT NULL DEFAULT '',
    email_verified BOOLEAN NOT NULL DEFAULT false,
    email_enabled BOOLEAN NOT NULL DEFAULT false,
    phone_number VARCHAR(50) NOT NULL DEFAULT '',
    phone_verified BOOLEAN NOT NULL DEFAULT false,
    sms_enabled BOOLEAN NOT NULL DEFAULT false,
    whatsapp_enabled BOOLEAN NOT NULL DEFAULT false
);
