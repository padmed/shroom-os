-- 2. Create climate_logs (Matches app/page.tsx exactly)
CREATE TABLE public.climate_logs (
    id BIGSERIAL PRIMARY KEY,
    device_id VARCHAR(50) NOT NULL DEFAULT 'hardware_simulation_node',
    temperature NUMERIC NOT NULL,
    humidity NUMERIC NOT NULL,
    co2_ppm INTEGER NOT NULL,
    recorded_at TIMESTAMPTZ DEFAULT NOW()
);

-- 3. Create device_controls (Matches app/page.tsx exactly)
CREATE TABLE public.device_controls (
    device_id VARCHAR(50) PRIMARY KEY,
    mode VARCHAR(20) NOT NULL DEFAULT 'AUTO'
);

-- 4. Create autopilot_status (Matches app/autopilot/page.tsx exactly)
CREATE TABLE public.autopilot_status (
    id INT PRIMARY KEY DEFAULT 1,
    strain_name VARCHAR(100) NOT NULL DEFAULT 'None',
    current_stage VARCHAR(50) NOT NULL DEFAULT 'OFF',
    started_at TIMESTAMPTZ -- <-- This was causing your 400 error!
);

-- 5. Create growth_blueprints (Matches app/autopilot/page.tsx exactly)
CREATE TABLE public.growth_blueprints (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

    incubation_humidity NUMERIC,
    incubation_co2 INT,
    pinning_days INT,
    pinning_temp NUMERIC,
    pinning_humidity NUMERIC,
    pinning_co2 INT,
    fruiting_days INT,
    fruiting_temp NUMERIC,
    fruiting_humidity NUMERIC,
    fruiting_co2 INT
);
