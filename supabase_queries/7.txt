CREATE TABLE public.growth_blueprints (
    id BIGSERIAL PRIMARY KEY,
    strain_name VARCHAR(100) NOT NULL UNIQUE,
    is_custom BOOLEAN DEFAULT TRUE,
    
    -- Phase 1: Incubation Targets
    incubation_days INT NOT NULL DEFAULT 14,
    incubation_temp NUMERIC(4,2) NOT NULL DEFAULT 24.0,
    incubation_humidity NUMERIC(4,2) NOT NULL DEFAULT 90.0,
    incubation_co2 INT NOT NULL DEFAULT 2000,
    
    -- Phase 2: Pinning Targets
    pinning_days INT NOT NULL DEFAULT 4,
    pinning_temp NUMERIC(4,2) NOT NULL DEFAULT 18.0,
    pinning_humidity NUMERIC(4,2) NOT NULL DEFAULT 95.0,
    pinning_co2 INT NOT NULL DEFAULT 800,
    
    -- Phase 3: Fruiting Targets
    fruiting_days INT NOT NULL DEFAULT 7,
    fruiting_temp NUMERIC(4,2) NOT NULL DEFAULT 21.0,
    fruiting_humidity NUMERIC(4,2) NOT NULL DEFAULT 85.0,
    fruiting_co2 INT NOT NULL DEFAULT 900,
    
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Seed the system defaults so the list isn't empty on bootup

VALUES

ON CONFLICT (strain_name) DO NOTHING;
